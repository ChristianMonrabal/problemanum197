Algoritmo Desencriptar_X2
    // Definir las variables para el texto del principio y el del final
    TextoPrincipio <- ''
    TextoFinal <- ''
    // Solicitar el texto a desencriptar
    Escribir 'Introduce el texto a desencriptar:'
    Leer Texto
    // Recorrer cada carácter del texto
    Para i<-1 Hasta longitud(Texto) Con Paso 1 Hacer
        // Si el índice es par, añadir al principio
        Si i MOD 2=0 Entonces
            TextoPrincipio <- TextoPrincipio+SubCadena(Texto,i,1)
            // Si el índice es impar, añadir al final
        SiNo
            TextoFinal <- SubCadena(Texto,i,1)+TextoFinal
        FinSi
    FinPara
    // Devolver la concatenación de los textos
    Escribir 'Texto desencriptado: ', TextoPrincipio+TextoFinal
FinAlgoritmo
